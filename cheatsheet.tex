\section{Graph algorithms}
\begin{algorithm}[H]
\BellmanFord{\Vertices,\Edges,\Source}{
  \For{$\V\in\Vertices$}{
    $\Distance[\V]\gets\infty$\;
    $\Predecessor[\V]\gets\Null$\;
  }
  $\Distance[\Source]\gets 0$\;
  \For{$i\leftarrow 1$ \KwTo $\abs{\Vertices}-1$}{
    \For{$\tupl{\U,\V,\Weight}\in\Edges$}{
      \If{$\Distance[\U]+\Weight <\Distance[\V]$}{
	$\Distance[\V]\gets\Distance[\U]+\Weight$\;
	$\Predecessor[\V]\gets\U$\;
      }
    }
  }
  \For{$\tupl{\U,\V,\Weight}\in\Edges$}{
    \If{$\Distance[\U]+\Weight < \Distance[\V]$}{
      \Error{"Graph negative-weight cycles contains a negative-weight cycle"}
    }
  }
}
\end{algorithm}
